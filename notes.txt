Windows cmds:
1️. Create Conda Environment 
conda --version
conda create -n trustcast python=3.12 -y
conda init
conda activate trustcast
pip install -r requirements.txt
pip install "zenml[server]"


2. Local Zenml setup:
zenml login --local --blocking
zenml init
zenml status

3. Install MLflow integration
zenml integration install mlflow -y
pip install "zenml[mlflow]"


Register MLflow experiment tracker
zenml experiment-tracker register trustcast_tracker --flavor=mlflow
zenml experiment-tracker list


7️⃣ Create and Activate Stack (✔ correct)

zenml stack register trustcast_stack \
  -a default \
  -o default \
  -e trustcast_tracker
zenml stack set trustcast_stack
zenml stack describe
zenml login --local --blocking

conda activate trustcast

frontend
cd frontend
npm install
npm run dev





Ubuntu cmds:
1. Create Environment
        conda --version
        conda create --name trustcast python=3.12
        conda init
        conda activate trustcast
2. Install required libraries
        pip install -r requirements.txt
3. version problem solution:  pip install pydantic-settings>=2.5.2 rich>=13.9.4 pygments>=2.4.1 traitlets>=5.3 ipython==8.12.3
4. Zenml setup: 
        pip install zenml["server"]
        zenml up
5. MLflow integrations:
        zenml integration install mlflow -y
        pip install zenml[mlflow]
        zenml integration install mlflow
        zenml init
6. Register MLflow as an experiment tracker
        zenml experiment-tracker register trustcast_tracker --flavor=mlflow
        zenml experiment-tracker list

        zenml stack register trustcast_stack -a local -o local -e trustcast_tracker
        zenml stack set trustcast_stack
        zenml stack describe

pull cmds:

To push Large File like .csv files
sudo apt-get install git-lfs
git lfs track "*.csv"
git add .gitattributes
git commit -m "Track CSV files with Git LFS"
git add data/*.csv --force
git commit -m "Move large CSVs to LFS"
git push
